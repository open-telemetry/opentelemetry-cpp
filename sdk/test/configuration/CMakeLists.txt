foreach(testname yaml_logs_test yaml_metrics_test yaml_propagator_test
                 yaml_resource_test yaml_test yaml_trace_test)

    TARGET ${testname}
    TEST_PREFIX trace.
    TEST_LIST ${testname})
endforeach()
