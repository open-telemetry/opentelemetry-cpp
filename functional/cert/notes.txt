cfssl genkey -initca ca_csr.json | cfssljson -bare ca

cfssl gencert -ca ca.pem -ca-key ca-key.pem client_csr.json | cfssljson -bare client_cert
cfssl gencert -ca ca.pem -ca-key ca-key.pem server_csr.json | cfssljson -bare server_cert

openssl x509 -in ca.pem -text
openssl x509 -in client_cert.pem -text
openssl x509 -in server_cert.pem -text

openssl verify -CAfile ca.pem server_cert.pem client_cert.pem

openssl s_server -accept 4318 -CAfile ca.pem -cert server_cert.pem -key server_cert-key.pem

openssl s_client -CAfile ca.pem -cert client_cert.pem -key client_cert-key.pem -connect 4318
