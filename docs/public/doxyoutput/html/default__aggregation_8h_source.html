23class AggregationConfig;

25class DefaultAggregation
26{
27public:
28 static std::unique_ptr<Aggregation> CreateAggregation(
29 const opentelemetry::sdk::metrics::InstrumentDescriptor &instrument_descriptor,
30 const AggregationConfig *aggregation_config)
31 {
32 bool is_monotonic = true;
33 auto aggr_type = GetDefaultAggregationType(instrument_descriptor.type_, is_monotonic);
34 switch (aggr_type)
35 {
36 case AggregationType::kSum:
37 return (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
38 ? std::move(std::unique_ptr<Aggregation>(new LongSumAggregation(is_monotonic)))
39 : std::move(
40 std::unique_ptr<Aggregation>(new DoubleSumAggregation(is_monotonic)));
41 break;
42 case AggregationType::kHistogram: {
43 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
44 {
45 return (std::unique_ptr<Aggregation>(new LongHistogramAggregation(aggregation_config)));
46 }
47 else
48 {
49 return (std::unique_ptr<Aggregation>(new DoubleHistogramAggregation(aggregation_config)));
50 }
51
52 break;
53 }
54 case AggregationType::kLastValue:
55 return (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
56 ? std::move(std::unique_ptr<Aggregation>(new LongLastValueAggregation()))
57 : std::move(std::unique_ptr<Aggregation>(new DoubleLastValueAggregation()));
58 break;
59 default:
60 return std::unique_ptr<Aggregation>(new DropAggregation());
61 };
62 }
63
64 static std::unique_ptr<Aggregation> CreateAggregation(
65 AggregationType aggregation_type,
66 const InstrumentDescriptor &instrument_descriptor,
67 const AggregationConfig *aggregation_config = nullptr)
68 {
69 switch (aggregation_type)
70 {
71 case AggregationType::kDrop:
72 return std::unique_ptr<Aggregation>(new DropAggregation());
73 break;
74 case AggregationType::kHistogram:
75 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
76 {
77 return std::unique_ptr<Aggregation>(new LongHistogramAggregation(aggregation_config));
78 }
79 else
80 {
81 return std::unique_ptr<Aggregation>(new DoubleHistogramAggregation(aggregation_config));
82 }
83 break;
84 case AggregationType::kBase2ExponentialHistogram:
85 return std::unique_ptr<Aggregation>(
86 new Base2ExponentialHistogramAggregation(aggregation_config));
87 break;
88 case AggregationType::kLastValue:
89 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
90 {
91 return std::unique_ptr<Aggregation>(new LongLastValueAggregation());
92 }
93 else
94 {
95 return std::unique_ptr<Aggregation>(new DoubleLastValueAggregation());
96 }
97 break;
98 case AggregationType::kSum: {
99 bool is_monotonic = true;
100 if (instrument_descriptor.type_ == InstrumentType::kUpDownCounter ||
101 instrument_descriptor.type_ == InstrumentType::kObservableUpDownCounter ||
102 instrument_descriptor.type_ == InstrumentType::kHistogram)
103 {
104 is_monotonic = false;
105 }
106 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
107 {
108 return std::unique_ptr<Aggregation>(new LongSumAggregation(is_monotonic));
109 }
110 else
111 {
112 return std::unique_ptr<Aggregation>(new DoubleSumAggregation(is_monotonic));
113 }
114 break;
115 }
116 default:
117 return DefaultAggregation::CreateAggregation(instrument_descriptor, aggregation_config);
118 }
119 }

121 static std::unique_ptr<Aggregation> CloneAggregation(
122 AggregationType aggregation_type,
123 const InstrumentDescriptor &instrument_descriptor,
124 const Aggregation &to_copy)
125 {
126 const PointType point_data = to_copy.ToPoint();
127 bool is_monotonic = true;
128 if (aggregation_type == AggregationType::kDefault)
129 {
130 aggregation_type = GetDefaultAggregationType(instrument_descriptor.type_, is_monotonic);
131 }
132 switch (aggregation_type)
133 {
134 case AggregationType::kDrop:
135 return std::unique_ptr<Aggregation>(new DropAggregation());
136 case AggregationType::kHistogram:
137 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
138 {
139 return std::unique_ptr<Aggregation>(
140 new LongHistogramAggregation(nostd::get<HistogramPointData>(point_data)));
141 }
142 else
143 {
144 return std::unique_ptr<Aggregation>(
145 new DoubleHistogramAggregation(nostd::get<HistogramPointData>(point_data)));
146 }
147 case AggregationType::kBase2ExponentialHistogram:
148 return std::unique_ptr<Aggregation>(new Base2ExponentialHistogramAggregation(
149 nostd::get<Base2ExponentialHistogramPointData>(point_data)));
150 case AggregationType::kLastValue:
151 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
152 {
153 return std::unique_ptr<Aggregation>(
154 new LongLastValueAggregation(nostd::get<LastValuePointData>(point_data)));
155 }
156 else
157 {
158 return std::unique_ptr<Aggregation>(
159 new DoubleLastValueAggregation(nostd::get<LastValuePointData>(point_data)));
160 }
161 case AggregationType::kSum:
162 if (instrument_descriptor.value_type_ == InstrumentValueType::kLong)
163 {
164 return std::unique_ptr<Aggregation>(
165 new LongSumAggregation(nostd::get<SumPointData>(point_data)));
166 }
167 else
168 {
169 return std::unique_ptr<Aggregation>(
170 new DoubleSumAggregation(nostd::get<SumPointData>(point_data)));
171 }
172 default:
173 return nullptr; // won't reach here
174 }
175 }
176
177 static AggregationType GetDefaultAggregationType(InstrumentType instrument_type,
178 bool &is_monotonic)
179 {
180 is_monotonic = false;
181 switch (instrument_type)
182 {
183 case InstrumentType::kCounter:
184 case InstrumentType::kObservableCounter:
185 is_monotonic = true;
186 return AggregationType::kSum;
187 case InstrumentType::kUpDownCounter:
188 case InstrumentType::kObservableUpDownCounter:
189 return AggregationType::kSum;
190 case InstrumentType::kHistogram:
191 return AggregationType::kHistogram;
192 case InstrumentType::kGauge:
193 case InstrumentType::kObservableGauge:
194 return AggregationType::kLastValue;
195 default:
196 return AggregationType::kDrop;
197 }
198 }
199};
