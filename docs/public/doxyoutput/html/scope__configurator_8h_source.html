79 ScopeConfigurator<T> Build() const
80 {
81 if (conditions_.size() == 0)
82 {
83 return ScopeConfigurator<T>(
84 [default_scope_config_ = this->default_scope_config_](const InstrumentationScope &) {
85 return default_scope_config_;
86 });
87 }
88
89 // Return a configurator that processes all the conditions
90 return ScopeConfigurator<T>(
91 [conditions_ = this->conditions_, default_scope_config_ = this->default_scope_config_](
92 const InstrumentationScope &scope_info) {
93 for (const Condition &condition : conditions_)
94 {
95 if (condition.scope_matcher(scope_info))
96 {
97 return condition.scope_config;
98 }
99 }
100 return default_scope_config_;
101 });
102 }
