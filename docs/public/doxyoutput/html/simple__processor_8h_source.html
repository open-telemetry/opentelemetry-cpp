91protected:
92 bool InternalShutdown(

94 {
95 // We only call shutdown ONCE.
96 if (exporter_ != nullptr && !shutdown_latch_.test_and_set(std::memory_order_acquire))
97 {
98 return exporter_->Shutdown(timeout);
99 }
100 return true;
101 }

106#if defined(__cpp_lib_atomic_value_initialization) && \
