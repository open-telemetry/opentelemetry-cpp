15// TraceFlags represents options for a Trace. These options are propagated to all child Spans
16// and determine features such as whether a Span should be traced. TraceFlags
17// are implemented as a bitmask.
18class TraceFlags final
19{
20public:
21 static constexpr uint8_t kIsSampled = 1;
22 static constexpr uint8_t kIsRandom = 2;
23
28 static constexpr uint8_t kAllW3CTraceContext1Flags = kIsSampled;
29
34 static constexpr uint8_t kAllW3CTraceContext2Flags = kIsSampled | kIsRandom;
35
36 TraceFlags() noexcept : rep_{0} {}
37
38 explicit TraceFlags(uint8_t flags) noexcept : rep_(flags) {}
39
40 bool IsSampled() const noexcept { return rep_ & kIsSampled; }
41
42 bool IsRandom() const noexcept { return rep_ & kIsRandom; }
43
44 // Populates the buffer with the lowercase base16 representation of the flags.
45 void ToLowerBase16(nostd::span<char, 2> buffer) const noexcept
46 {
47 constexpr char kHex[] = "0123456789ABCDEF";
48 buffer[0] = kHex[(rep_ >> 4) & 0xF];
49 buffer[1] = kHex[(rep_ >> 0) & 0xF];
50 }
51
52 uint8_t flags() const noexcept { return rep_; }
53
54 bool operator==(const TraceFlags &that) const noexcept { return rep_ == that.rep_; }
55
56 bool operator!=(const TraceFlags &that) const noexcept { return !(*this == that); }
57
58 // Copies the TraceFlags to dest.
59 void CopyBytesTo(nostd::span<uint8_t, 1> dest) const noexcept { dest[0] = rep_; }
60
61private:
62 uint8_t rep_;
63};
64
