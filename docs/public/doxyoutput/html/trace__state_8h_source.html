55 static nostd::shared_ptr<TraceState> FromHeader(nostd::string_view header) noexcept
56 {
57
58 common::KeyValueStringTokenizer kv_str_tokenizer(header);
59 size_t cnt = kv_str_tokenizer.NumTokens(); // upper bound on number of kv pairs
60 if (cnt > kMaxKeyValuePairs)
61 {
62 // trace state should be discarded if count exceeds
63 return GetDefault();
64 }
65
66 nostd::shared_ptr<TraceState> ts(new TraceState(cnt));
67 bool kv_valid{false};
68 nostd::string_view key, value;
69 while (kv_str_tokenizer.next(kv_valid, key, value) && ts->kv_properties_->Size() < cnt)
70 {
71 if (kv_valid == false)
72 {
73 return GetDefault();
74 }
75
76 if (!IsValidKey(key) || !IsValidValue(value))
77 {
78 // invalid header. return empty TraceState
79 ts->kv_properties_.reset(new common::KeyValueProperties());
80 break;
81 }
82
83 ts->kv_properties_->AddEntry(key, value);
84 }
85
86 return ts;
87 }

92 std::string ToHeader() const noexcept
93 {
94 std::string header_s;
95 bool first = true;
96 kv_properties_->GetAllEntries(
97 [&header_s, &first](nostd::string_view key, nostd::string_view value) noexcept {
98 if (!first)
99 {
100 header_s.append(1, kMembersSeparator);
101 }
102 else
103 {
104 first = false;
105 }
106 header_s.append(std::string(key.data(), key.size()));
107 header_s.append(1, kKeyValueSeparator);
108 header_s.append(std::string(value.data(), value.size()));
109 return true;
110 });
111 return header_s;
112 }

138 nostd::shared_ptr<TraceState> Set(const nostd::string_view &key,
139 const nostd::string_view &value) noexcept
140 {
141 auto curr_size = kv_properties_->Size();
142 if (!IsValidKey(key) || !IsValidValue(value))
143 {
144 // max size reached or invalid key/value. Returning empty TraceState
145 return TraceState::GetDefault();
146 }
147 auto allocate_size = curr_size;
148 if (curr_size < kMaxKeyValuePairs)
149 {
150 allocate_size += 1;
151 }
152 nostd::shared_ptr<TraceState> ts(new TraceState(allocate_size));
153 if (curr_size < kMaxKeyValuePairs)
154 {
155 // add new field first
156 ts->kv_properties_->AddEntry(key, value);
157 }
158 // add rest of the fields.
159 kv_properties_->GetAllEntries([&ts](nostd::string_view key, nostd::string_view value) {
160 ts->kv_properties_->AddEntry(key, value);
161 return true;
162 });
163 return ts;
164 }

172 nostd::shared_ptr<TraceState> Delete(const nostd::string_view &key) noexcept
173 {
174 if (!IsValidKey(key))
175 {
176 return TraceState::GetDefault();
177 }
178 auto curr_size = kv_properties_->Size();
179 auto allocate_size = curr_size;
180 std::string unused;
181 if (kv_properties_->GetValue(key, unused))
182 {
183 allocate_size -= 1;
184 }
185 nostd::shared_ptr<TraceState> ts(new TraceState(allocate_size));
186 kv_properties_->GetAllEntries(
187 [&ts, &key](nostd::string_view e_key, nostd::string_view e_value) {
188 if (key != e_key)
189 ts->kv_properties_->AddEntry(e_key, e_value);
190 return true;
191 });
192 return ts;
193 }

238 static nostd::string_view TrimString(nostd::string_view str, size_t left, size_t right)
239 {
240 while (str[static_cast<std::size_t>(right)] == ' ' && left < right)
241 {
242 right--;
243 }
244 while (str[static_cast<std::size_t>(left)] == ' ' && left < right)
245 {
246 left++;
247 }
248 return str.substr(left, right - left + 1);
249 }

251#if OPENTELEMETRY_HAVE_WORKING_REGEX
252 static bool IsValidKeyRegEx(nostd::string_view key)
253 {
254 static std::regex reg_key("^[a-z0-9][a-z0-9*_\\-/]{0,255}$");
255 static std::regex reg_key_multitenant(
256 "^[a-z0-9][a-z0-9*_\\-/]{0,240}(@)[a-z0-9][a-z0-9*_\\-/]{0,13}$");
257 std::string key_s(key.data(), key.size());
258 if (std::regex_match(key_s, reg_key) || std::regex_match(key_s, reg_key_multitenant))
259 {
260 return true;
261 }
262 return false;
263 }
264
265 static bool IsValidValueRegEx(nostd::string_view value)
266 {
267 // Hex 0x20 to 0x2B, 0x2D to 0x3C, 0x3E to 0x7E
268 static std::regex reg_value(
269 "^[\\x20-\\x2B\\x2D-\\x3C\\x3E-\\x7E]{0,255}[\\x21-\\x2B\\x2D-\\x3C\\x3E-\\x7E]$");
270 // Need to benchmark without regex, as a string object is created here.
271 return std::regex_match(std::string(value.data(), value.size()), reg_value);
272 }
273#else
274 static bool IsValidKeyNonRegEx(nostd::string_view key)
275 {
276 if (key.empty() || key.size() > kKeyMaxSize || !IsLowerCaseAlphaOrDigit(key[0]))
277 {
278 return false;
279 }
280
281 int ats = 0;
282
283 for (const char c : key)
284 {
285 if (!IsLowerCaseAlphaOrDigit(c) && c != '_' && c != '-' && c != '@' && c != '*' && c != '/')
286 {
287 return false;
288 }
289 if ((c == '@') && (++ats > 1))
290 {
291 return false;
292 }
293 }
294 return true;
295 }
296
297 static bool IsValidValueNonRegEx(nostd::string_view value)
298 {
299 if (value.empty() || value.size() > kValueMaxSize)
300 {
301 return false;
302 }
303
304 for (const char c : value)
305 {
306 if (c < ' ' || c > '~' || c == ',' || c == '=')
307 {
308 return false;
309 }
310 }
311 return true;
312 }
313#endif
314
315 static bool IsLowerCaseAlphaOrDigit(char c) { return isdigit(c) || islower(c); }
316
317private:
318 // Store entries in a C-style array to avoid using std::array or std::vector.
319 nostd::unique_ptr<common::KeyValueProperties> kv_properties_;
