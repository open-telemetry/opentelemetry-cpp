21// RuntimeContext object and is associated with a context object, and
22// can be provided to the RuntimeContext Detach method to remove the

100// Provides a wrapper for propagating the context object globally.
101//
102// By default, a thread-local runtime context storage is used.
103class OPENTELEMETRY_EXPORT RuntimeContext
104{
105public:
106 // Return the current context.
107 static Context GetCurrent() noexcept { return GetRuntimeContextStorage()->GetCurrent(); }
108
109 // Sets the current 'Context' object. Returns a token
110 // that can be used to reset to the previous Context.
111 static nostd::unique_ptr<Token> Attach(const Context &context) noexcept
112 {
113 return GetRuntimeContextStorage()->Attach(context);
114 }
115
116 // Resets the context to a previous value stored in the
117 // passed in token. Returns true if successful, false otherwise
118 static bool Detach(Token &token) noexcept { return GetRuntimeContextStorage()->Detach(token); }
119
120 // Sets the Key and Value into the passed in context or if a context is not
121 // passed in, the RuntimeContext.
122 // Should be used to SetValues to the current RuntimeContext, is essentially
123 // equivalent to RuntimeContext::GetCurrent().SetValue(key,value). Keep in
124 // mind that the current RuntimeContext will not be changed, and the new
125 // context will be returned.
126 static Context SetValue(nostd::string_view key,
127 const ContextValue &value,
128 Context *context = nullptr) noexcept
129 {
130 Context temp_context;
131 if (context == nullptr)
132 {
133 temp_context = GetCurrent();
134 }
135 else
136 {
137 temp_context = *context;
138 }
139 return temp_context.SetValue(key, value);
140 }
141
142 // Returns the value associated with the passed in key and either the
143 // passed in context* or the runtime context if a context is not passed in.
144 // Should be used to get values from the current RuntimeContext, is
145 // essentially equivalent to RuntimeContext::GetCurrent().GetValue(key).
146 static ContextValue GetValue(nostd::string_view key, Context *context = nullptr) noexcept
147 {
148 Context temp_context;
149 if (context == nullptr)
150 {
151 temp_context = GetCurrent();
152 }
153 else
154 {
155 temp_context = *context;
156 }
157 return temp_context.GetValue(key);
158 }

205// The ThreadLocalContextStorage class is a derived class from
206// RuntimeContextStorage and provides a wrapper for propagating context through
207// cpp thread locally. This file must be included to use the RuntimeContext
208// class if another implementation has not been registered.
209class ThreadLocalContextStorage : public RuntimeContextStorage
210{
211public:
212 ThreadLocalContextStorage() noexcept = default;
213
214 // Return the current context.
215 Context GetCurrent() noexcept override { return GetStack().Top(); }
216
217 // Resets the context to the value previous to the passed in token. This will
218 // also detach all child contexts of the passed in token.
219 // Returns true if successful, false otherwise.
220 bool Detach(Token &token) noexcept override
221 {
222 // In most cases, the context to be detached is on the top of the stack.
223 if (token == GetStack().Top())
224 {
225 GetStack().Pop();
226 return true;
227 }
228
229 if (!GetStack().Contains(token))
230 {
231 return false;
232 }
233
234 while (!(token == GetStack().Top()))
235 {
236 GetStack().Pop();
237 }
238
239 GetStack().Pop();
240
241 return true;
242 }
243
244 // Sets the current 'Context' object. Returns a token
245 // that can be used to reset to the previous Context.
246 nostd::unique_ptr<Token> Attach(const Context &context) noexcept override
247 {
248 GetStack().Push(context);
249 return CreateToken(context);
250 }
251
252private:
253 // A nested class to store the attached contexts in a stack.
254 class Stack
255 {
256 friend class ThreadLocalContextStorage;
257
276 static constexpr size_t max_capacity_ = 1000000;
277
278 Stack() noexcept = default;
279 Stack(const Stack &) = delete;
280 Stack &operator=(const Stack &) = delete;
281 Stack(Stack &&) = delete;
282 Stack &operator=(Stack &&) = delete;
283
284 // Pops the top Context off the stack.
285 void Pop() noexcept
286 {
287 if (size_ == 0)
288 {
289 return;
290 }
291 if (size_ > max_capacity_)
292 {
293 size_ -= 1;
294 return;
295 }
296 // Store empty Context before decrementing `size`, to ensure
297 // the shared_ptr object (if stored in prev context object ) are released.
298 // The stack is not resized, and the unused memory would be reutilised
299 // for subsequent context storage.
300 base_[size_ - 1] = Context();
301 size_ -= 1;
302 }
303
304 bool Contains(const Token &token) const noexcept
305 {
306 for (size_t pos = size_; pos > 0; --pos)
307 {
308 if (token == base_[pos - 1])
309 {
310 return true;
311 }
312 }
313
314 return false;
315 }
316
317 // Returns the Context at the top of the stack.
318 Context Top() const noexcept
319 {
320 if (size_ == 0)
321 {
322 return Context();
323 }
324 if (size_ > max_capacity_)
325 {
326 return Context();
327 }
328 return base_[size_ - 1];
329 }
330
331 // Pushes the passed in context pointer to the top of the stack
332 // and resizes if necessary.
333 void Push(const Context &context) noexcept
334 {
335 size_++;
336 if (size_ > max_capacity_)
337 {
338 return;
339 }
340 if (size_ > capacity_)
341 {
342 Resize(size_ * 2);
343 }
344 base_[size_ - 1] = context;
345 }
346
347 // Reallocates the storage array to the pass in new capacity size.
348 void Resize(size_t new_capacity) noexcept
349 {
350 size_t old_size = size_ - 1;
351 if (new_capacity == 0)
352 {
353 // First increase
354 new_capacity = 2;
355 }
356 if (new_capacity > max_capacity_)
357 {
358 // Last increase
359 new_capacity = max_capacity_;
360 }
361 Context *temp = new Context[new_capacity];
362 if (base_ != nullptr)
363 {
364 // vs2015 does not like this construct considering it unsafe:
365 // - std::copy(base_, base_ + old_size, temp);
366 // Ref.
367 // https://stackoverflow.com/questions/12270224/xutility2227-warning-c4996-std-copy-impl
368 for (size_t i = 0; i < (std::min)(old_size, new_capacity); i++)
369 {
370 temp[i] = base_[i];
371 }
372 delete[] base_;
373 }
374 base_ = temp;
375 capacity_ = new_capacity;
376 }
377
378 ~Stack() noexcept { delete[] base_; }
379
380 size_t size_{};
381 size_t capacity_{};
382 Context *base_{nullptr};
383 };
384
385 OPENTELEMETRY_API_SINGLETON Stack &GetStack()
386 {
387 static thread_local Stack stack_{};
388 return stack_;
389 }
390};
