38enum class Decision : std::uint8_t
39{
40 // IsRecording() == false, span will not be recorded and all events and attributes will be
41 // dropped.
42 DROP,
43 // IsRecording() == true, but Sampled flag MUST NOT be set.
44 RECORD_ONLY,
45 // IsRecording() == true AND Sampled flag` MUST be set.
46 RECORD_AND_SAMPLE
47};

53struct SamplingResult
54{
55 Decision decision;
56 // A set of span Attributes that will also be added to the Span. Can be nullptr.
57 std::unique_ptr<const std::map<std::string, opentelemetry::common::AttributeValue>> attributes;
58 // The tracestate used by the span.
59 nostd::shared_ptr<opentelemetry::trace::TraceState> trace_state;
60
61 inline bool IsRecording()
62 {
63 return decision == Decision::RECORD_ONLY || decision == Decision::RECORD_AND_SAMPLE;
64 }
65 inline bool IsSampled() { return decision == Decision::RECORD_AND_SAMPLE; }
66};
