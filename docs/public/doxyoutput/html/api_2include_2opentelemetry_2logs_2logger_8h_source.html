78 template <class... ArgumentType>
79 void EmitLogRecord(nostd::unique_ptr<LogRecord> &&log_record, ArgumentType &&...args)
80 {
81 if (!log_record)
82 {
83 return;
84 }
85
86 //
87 // Keep the parameter pack unpacking order from left to right because left
88 // ones are usually more important like severity and event_id than the
89 // attributes. The left to right unpack order could pass the more important
90 // data to processors to avoid caching and memory allocating.
91 //
92#if __cplusplus <= 201402L
93 // C++14 does not support fold expressions for parameter pack expansion.
94 int dummy[] = {(detail::LogRecordSetterTrait<typename std::decay<ArgumentType>::type>::Set(
95 log_record.get(), std::forward<ArgumentType>(args)),
96 0)...};
97 IgnoreTraitResult(dummy);
98#else
99 IgnoreTraitResult((detail::LogRecordSetterTrait<typename std::decay<ArgumentType>::type>::Set(
100 log_record.get(), std::forward<ArgumentType>(args)),
101 ...));
102#endif
103
104 EmitLogRecord(std::move(log_record));
105 }
106
124 template <class... ArgumentType>
125 void EmitLogRecord(ArgumentType &&...args)
126 {
127 nostd::unique_ptr<LogRecord> log_record = CreateLogRecord();
128
129 EmitLogRecord(std::move(log_record), std::forward<ArgumentType>(args)...);
130 }
