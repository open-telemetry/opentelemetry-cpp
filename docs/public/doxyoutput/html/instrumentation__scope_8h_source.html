79 template <
80 class ArgumentType,
81 nostd::enable_if_t<opentelemetry::common::detail::is_key_value_iterable<ArgumentType>::value>
82 * = nullptr>
83 static nostd::unique_ptr<InstrumentationScope> Create(nostd::string_view name,
84 nostd::string_view version,
85 nostd::string_view schema_url,
86 const ArgumentType &arg)
87 {
88 nostd::unique_ptr<InstrumentationScope> result = nostd::unique_ptr<InstrumentationScope>(
89 new InstrumentationScope{name, version, schema_url});
90
91 // Do not construct a KeyValueIterable, so it has better performance.
92 result->attributes_.reserve(opentelemetry::nostd::size(arg));
93 for (auto &argv : arg)
94 {
95 result->SetAttribute(argv.first, argv.second);
96 }
97
98 return result;
99 }

124 bool equal(const nostd::string_view name,
125 const nostd::string_view version,
126 const nostd::string_view schema_url = "",
127 const opentelemetry::common::KeyValueIterable *attributes = nullptr) const noexcept
128 {
129
130 if (this->name_ != name || this->version_ != version || this->schema_url_ != schema_url)
131 {
132 return false;
133 }
134
135 if (attributes == nullptr)
136 {
137 if (attributes_.empty())
138 {
139 return true;
140 }
141 return false;
142 }
143
144 return attributes_.EqualTo(*attributes);
145 }

152 void SetAttribute(nostd::string_view key,
153 const opentelemetry::common::AttributeValue &value) noexcept
154 {
155 attributes_[std::string(key)] =
156 nostd::visit(opentelemetry::sdk::common::AttributeConverter(), value);
157 }
158
159private:
160 InstrumentationScope(nostd::string_view name,
161 nostd::string_view version,
162 nostd::string_view schema_url = "",
163 InstrumentationScopeAttributes &&attributes = {})
164 : name_(name), version_(version), schema_url_(schema_url), attributes_(std::move(attributes))
165 {
166 std::string hash_data;
167 hash_data.reserve(name_.size() + version_.size() + schema_url_.size());
168 hash_data += name_;
169 hash_data += version_;
170 hash_data += schema_url_;
171 hash_code_ = std::hash<std::string>{}(hash_data);
172 }
173
174private:
175 std::string name_;
176 std::string version_;
177 std::string schema_url_;
178 std::size_t hash_code_;
179
180 InstrumentationScopeAttributes attributes_;
181};
