81class FilteringAttributesProcessor : public AttributesProcessor
82{
83public:
84 FilteringAttributesProcessor(FilterAttributeMap &&allowed_attribute_keys = {})
85 : allowed_attribute_keys_(std::move(allowed_attribute_keys))
86 {}
87
88 FilteringAttributesProcessor(const FilterAttributeMap &allowed_attribute_keys = {})
89 : allowed_attribute_keys_(allowed_attribute_keys)
90 {}
91
92 MetricAttributes process(
93 const opentelemetry::common::KeyValueIterable &attributes) const noexcept override
94 {
95 MetricAttributes result;
96 attributes.ForEachKeyValue(
97 [&](nostd::string_view key, opentelemetry::common::AttributeValue value) noexcept {
98 if (opentelemetry::sdk::common::find_heterogeneous(allowed_attribute_keys_, key) !=
99 allowed_attribute_keys_.end())
100 {
101 result.SetAttribute(key, value);
102 return true;
103 }
104 return true;
105 });
106
107 result.UpdateHash();
108 return result;
109 }
110
111 bool isPresent(nostd::string_view key) const noexcept override
112 {
113 return (opentelemetry::sdk::common::find_heterogeneous(allowed_attribute_keys_, key) !=
114 allowed_attribute_keys_.end());
115 }
116
117private:
118 FilterAttributeMap allowed_attribute_keys_;
119};
