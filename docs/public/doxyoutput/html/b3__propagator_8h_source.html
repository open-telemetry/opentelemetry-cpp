21static const nostd::string_view kB3CombinedHeader = "b3";
22
23static const nostd::string_view kB3TraceIdHeader = "X-B3-TraceId";
24static const nostd::string_view kB3SpanIdHeader = "X-B3-SpanId";
25static const nostd::string_view kB3SampledHeader = "X-B3-Sampled";
26
27/*
28 B3, single header:
29 b3: 80f198ee56343ba864fe8b2a57d3eff7-e457b5a2e4d86bd1-1-05e3ac9a4f6e3b90
30 ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^ ^^^^^^^^^^^^^^^^ ^ ^^^^^^^^^^^^^^^^
31 0 TraceId 31 33 SpanId 48 | 52 ParentSpanId 68
32 50 Debug flag
33 Multiheader version: X-B3-Sampled
34 X-B3-TraceId X-B3-SpanId X-B3-ParentSpanId (ignored)
35*/
36
37static const int kTraceIdHexStrLength = 32;
38static const int kSpanIdHexStrLength = 16;
39
40// The B3PropagatorExtractor class provides an interface that enables extracting context from
41// headers of HTTP requests. HTTP frameworks and clients can integrate with B3Propagator by
42// providing the object containing the headers, and a getter function for the extraction. Based on:
43// https://github.com/open-telemetry/opentelemetry-specification/blob/main/specification/context/api-propagators.md#b3-extract
44
45class B3PropagatorExtractor : public context::propagation::TextMapPropagator
46{
47public:
48 // Returns the context that is stored in the HTTP header carrier.
49 context::Context Extract(const context::propagation::TextMapCarrier &carrier,
50 context::Context &context) noexcept override
51 {
52 SpanContext span_context = ExtractImpl(carrier);
53 nostd::shared_ptr<Span> sp{new DefaultSpan(span_context)};
54 if (span_context.IsValid())
55 {
56 return trace::SetSpan(context, sp);
57 }
58 else
59 {
60 return context;
61 }
62 }
63
64 static TraceId TraceIdFromHex(nostd::string_view trace_id)
65 {
66 uint8_t buf[kTraceIdHexStrLength / 2];
67 detail::HexToBinary(trace_id, buf, sizeof(buf));
68 return TraceId(buf);
69 }
70
71 static SpanId SpanIdFromHex(nostd::string_view span_id)
72 {
73 uint8_t buf[kSpanIdHexStrLength / 2];
74 detail::HexToBinary(span_id, buf, sizeof(buf));
75 return SpanId(buf);
76 }
77
78 static TraceFlags TraceFlagsFromHex(nostd::string_view trace_flags)
79 {
80 if (trace_flags.length() != 1 || (trace_flags[0] != '1' && trace_flags[0] != 'd'))
81 { // check for invalid length of flags and treat 'd' as sampled
82 return TraceFlags(0);
83 }
84 return TraceFlags(TraceFlags::kIsSampled);
85 }
86
87private:
88 static SpanContext ExtractImpl(const context::propagation::TextMapCarrier &carrier)
89 {
90 nostd::string_view trace_id_hex;
91 nostd::string_view span_id_hex;
92 nostd::string_view trace_flags_hex;
93
94 // first let's try a single-header variant
95 auto singleB3Header = carrier.Get(kB3CombinedHeader);
96 if (!singleB3Header.empty())
97 {
98 std::array<nostd::string_view, 3> fields{};
99 // https://github.com/openzipkin/b3-propagation/blob/master/RATIONALE.md
100 if (detail::SplitString(singleB3Header, '-', fields.data(), 3) < 2)
101 {
102 return SpanContext::GetInvalid();
103 }
104
105 trace_id_hex = fields[0];
106 span_id_hex = fields[1];
107 trace_flags_hex = fields[2];
108 }
109 else
110 {
111 trace_id_hex = carrier.Get(kB3TraceIdHeader);
112 span_id_hex = carrier.Get(kB3SpanIdHeader);
113 trace_flags_hex = carrier.Get(kB3SampledHeader);
114 }
115
116 if (!detail::IsValidHex(trace_id_hex) || !detail::IsValidHex(span_id_hex))
117 {
118 return SpanContext::GetInvalid();
119 }
120
121 TraceId trace_id = TraceIdFromHex(trace_id_hex);
122 SpanId span_id = SpanIdFromHex(span_id_hex);
123
124 if (!trace_id.IsValid() || !span_id.IsValid())
125 {
126 return SpanContext::GetInvalid();
127 }
128
129 return SpanContext(trace_id, span_id, TraceFlagsFromHex(trace_flags_hex), true);
130 }
131};

133// The B3Propagator class provides interface that enables extracting and injecting context into
134// single header of HTTP Request.
135class B3Propagator : public B3PropagatorExtractor
136{
137public:
138 // Sets the context for a HTTP header carrier with self defined rules.
139 void Inject(context::propagation::TextMapCarrier &carrier,
140 const context::Context &context) noexcept override
141 {
142 SpanContext span_context = trace::GetSpan(context)->GetContext();
143 if (!span_context.IsValid())
144 {
145 return;
146 }
147
148 char trace_identity[kTraceIdHexStrLength + kSpanIdHexStrLength + 3];
149 static_assert(sizeof(trace_identity) == 51, "b3 trace identity buffer size mismatch");
150 span_context.trace_id().ToLowerBase16(nostd::span<char, 2 * TraceId::kSize>{
151 &trace_identity[0], static_cast<std::size_t>(kTraceIdHexStrLength)});
152 trace_identity[kTraceIdHexStrLength] = '-';
153 span_context.span_id().ToLowerBase16(nostd::span<char, 2 * SpanId::kSize>{
154 &trace_identity[kTraceIdHexStrLength + 1], static_cast<std::size_t>(kSpanIdHexStrLength)});
155 trace_identity[kTraceIdHexStrLength + kSpanIdHexStrLength + 1] = '-';
156 trace_identity[kTraceIdHexStrLength + kSpanIdHexStrLength + 2] =
157 span_context.trace_flags().IsSampled() ? '1' : '0';
158
159 carrier.Set(kB3CombinedHeader, nostd::string_view(trace_identity, sizeof(trace_identity)));
160 }
161
162 bool Fields(nostd::function_ref<bool(nostd::string_view)> callback) const noexcept override
163 {
164 return callback(kB3CombinedHeader);
165 }
166};

168class B3PropagatorMultiHeader : public B3PropagatorExtractor
169{
170public:
171 void Inject(context::propagation::TextMapCarrier &carrier,
172 const context::Context &context) noexcept override
173 {
174 SpanContext span_context = GetSpan(context)->GetContext();
175 if (!span_context.IsValid())
176 {
177 return;
178 }
179 char trace_id[32];
180 TraceId(span_context.trace_id()).ToLowerBase16(trace_id);
181 char span_id[16];
182 SpanId(span_context.span_id()).ToLowerBase16(span_id);
183 char trace_flags[2];
184 TraceFlags(span_context.trace_flags()).ToLowerBase16(trace_flags);
185 carrier.Set(kB3TraceIdHeader, nostd::string_view(trace_id, sizeof(trace_id)));
186 carrier.Set(kB3SpanIdHeader, nostd::string_view(span_id, sizeof(span_id)));
187 carrier.Set(kB3SampledHeader, nostd::string_view(trace_flags + 1, 1));
188 }
189
190 bool Fields(nostd::function_ref<bool(nostd::string_view)> callback) const noexcept override
191 {
192 return callback(kB3TraceIdHeader) && callback(kB3SpanIdHeader) && callback(kB3SampledHeader);
193 }
194};
