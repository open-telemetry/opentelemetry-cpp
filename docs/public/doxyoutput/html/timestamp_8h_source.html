172class DurationUtil
173{
174public:
175 template <class Rep, class Period>
176 static std::chrono::duration<Rep, Period> AdjustWaitForTimeout(
177 std::chrono::duration<Rep, Period> timeout,
178 std::chrono::duration<Rep, Period> indefinite_value) noexcept
179 {
180 // Do not call now() when this duration is max value, now() may have a expensive cost.
181 if (timeout == (std::chrono::duration<Rep, Period>::max)())
182 {
183 return indefinite_value;
184 }
185
186 // std::future<T>::wait_for, std::this_thread::sleep_for, and std::condition_variable::wait_for
187 // may use steady_clock or system_clock.We need make sure now() + timeout do not overflow.
188 auto max_timeout = std::chrono::duration_cast<std::chrono::duration<Rep, Period>>(
189 (std::chrono::steady_clock::time_point::max)() - std::chrono::steady_clock::now());
190 if (timeout >= max_timeout)
191 {
192 return indefinite_value;
193 }
194 max_timeout = std::chrono::duration_cast<std::chrono::duration<Rep, Period>>(
195 (std::chrono::system_clock::time_point::max)() - std::chrono::system_clock::now());
196 if (timeout >= max_timeout)
197 {
198 return indefinite_value;
199 }
200
201 return timeout;
202 }
203};
